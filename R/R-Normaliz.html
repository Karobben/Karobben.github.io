Data Normalization in R

Karobben
20 Jan 2020

1 Max-Min


R 

for(ii in 1:ncol(a)){
    a[,ii] <- (a[,ii]-min(a[,ii]))/(max(a[,ii])-min(a[,ii]))
}


2 Z-score


R 

for(ii in 1:ncol(a)){
    a[,ii] <- (a[,ii]-mean(a[,ii]))/sd(a[,ii])
}


3 Soft-max Normalization


R 

for(ii in 1:ncol(a)){
    a[,ii] <- pnorm((a[,ii]-mean(a[,ii]))/sd(a[,ii]))
}


4 Logistic/Softmax


R 
for(ii in 1:ncol(a)){
    a[,ii] <- 1/(1+ exp(-a[,ii]))
}


5 模糊量化模式


R 
for(ii in 1:ncol(a)){
    x<- a[,ii]
    a[,ii] <- 1/2 +1/2sin(pi/(max(x)-mean(x))(x-max(x)-mean(x)/2)) *x
}


6 log


R 
for(ii in 1:ncol(a)){
a[,ii][which(a[,ii]>0)] <- log(a[,ii][which(a[,ii]>0)],10)/log(max(a[,ii]),10)
a[,ii][which(a[,ii]<0)] <- log(a[,ii][which(a[,ii]<0)](-1),10)(-1)/log(mean(a[,ii]) *(-1))
}


7 atan


R 
for(ii in 1:ncol(a)){
a[,ii][which(a[,ii]>0)] <- atan(a[,ii][which(a[,ii]>0)])*2/pi
a[,ii][which(a[,ii]<0)] <- atan(a[,ii][which(a[,ii]<0)])2/pi (-1)
}
